00034   CGColorSpaceRef space = CGColorSpaceCreateDeviceRGB();
00035   int bitmapBytesPerRow = (rect.size.width * 4);
00036   int bitmapByteCount = (bitmapBytesPerRow * rect.size.height);
00037   _maskData = malloc(bitmapByteCount);

00050 - (CGImageRef)newSpotlightMask:(CGRect)rect origin:(CGPoint)origin radius:(CGFloat)radius {
00051   CGContextClearRect(_maskContext, rect);
00052 
00053   CGColorSpaceRef space = CGColorSpaceCreateDeviceRGB();
00054   CGFloat components[] = {1, 1, 1, 1, 0, 0, 0, 0};
00055   CGFloat locations[] = {0, 1};
00056   CGGradientRef gradient = CGGradientCreateWithColorComponents(space, components, locations, 2);
00057   CGContextDrawRadialGradient(_maskContext, gradient, origin, 0, origin, radius, 0);
00058   CGGradientRelease(gradient);
00059   CGColorSpaceRelease(space);
00060 
00061   return CGBitmapContextCreateImage(_maskContext);
00062 }

00104 - (void)drawRect:(CGRect)rect {
00105   CGContextRef context = UIGraphicsGetCurrentContext();
00106 
00107   CGSize textSize = [self sizeThatFits:CGSizeZero];
00108   
00109   CGFloat x = 0;
00110   if (_textAlignment == UITextAlignmentRight) {
00111     x = self.frame.size.width - textSize.width;
00112   } else if (_textAlignment == UITextAlignmentCenter) {
00113     x = ceil(self.frame.size.width/2 - textSize.width/2);
00114   }
00115   
00116   CGFloat y = 0;
00117   if (self.contentMode == UIViewContentModeCenter) {
00118     y = ceil(rect.size.height/2 + _font.capHeight/2);
00119   } else if (self.contentMode == UIViewContentModeBottom) {
00120     y = rect.size.height + _font.descender;
00121   } else {
00122     y = _font.capHeight;
00123   }
00124   
00125   CGContextSelectFont(context, [_font.fontName UTF8String], _font.pointSize, kCGEncodingMacRoman);
00126   CGContextSetTextDrawingMode(context, kCGTextFill);
00127   CGContextSetTextMatrix(context, CGAffineTransformScale(CGAffineTransformIdentity, 1, -1));
00128 
00129   CGContextSetFillColorWithColor(context, textColor.CGColor);
00130   CGContextShowTextAtPoint(context, x, y, [self.text UTF8String], self.text.length);
00131 
00132   if (_timer) {
00133     CGPoint spotOrigin = CGPointMake(x + (textSize.width * _spotlightPoint),
00134       y - ceil(self.font.capHeight/2));
00135     CGFloat spotRadius = self.font.capHeight*2;
00136 
00137     CGImageRef mask = [self newSpotlightMask:rect origin:spotOrigin radius:spotRadius];
00138     CGContextClipToMask(context, rect, mask);
00139     CGImageRelease(mask);
00140     
00141     [_spotlightColor setFill];
00142     CGContextShowTextAtPoint(context, x, y, [self.text UTF8String], self.text.length);
00143   }
00144 }
