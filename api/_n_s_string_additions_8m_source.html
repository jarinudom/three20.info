00138   NSMutableArray* pairs = [NSMutableArray array];
00139   for (NSString* key in [query keyEnumerator]) {
00140     NSString* value = [query objectForKey:key];
00141     value = [value stringByReplacingOccurrencesOfString:@"?" withString:@"%3F"];
00142     value = [value stringByReplacingOccurrencesOfString:@"=" withString:@"%3D"];
00143     NSString* pair = [NSString stringWithFormat:@"%@=%@", key, value];
00144     [pairs addObject:pair];

00168 - (NSComparisonResult)versionStringCompare:(NSString *)other {
00169   NSArray *oneComponents = [self componentsSeparatedByString:@"a"];
00170   NSArray *twoComponents = [other componentsSeparatedByString:@"a"];
00171 
00172   // The parts before the "a"
00173   NSString *oneMain = [oneComponents objectAtIndex:0];
00174   NSString *twoMain = [twoComponents objectAtIndex:0];
00175 
00176   // If main parts are different, return that result, regardless of alpha part
00177   NSComparisonResult mainDiff;
00178   if ((mainDiff = [oneMain compare:twoMain]) != NSOrderedSame) {
00179     return mainDiff;
00180   }
00181 
00182   // At this point the main parts are the same; just deal with alpha stuff
00183   // If one has an alpha part and the other doesn't, the one without is newer
00184   if ([oneComponents count] < [twoComponents count]) {
00185     return NSOrderedDescending;
00186   } else if ([oneComponents count] > [twoComponents count]) {
00187     return NSOrderedAscending;
00188   } else if ([oneComponents count] == 1) {
00189     // Neither has an alpha part, and we know the main parts are the same
00190     return NSOrderedSame;
00191   }
00192 
00193   // At this point the main parts are the same and both have alpha parts. Compare the alpha parts
00194   // numerically. If it's not a valid number (including empty string) it's treated as zero.
00195   NSNumber *oneAlpha = [NSNumber numberWithInt:[[oneComponents objectAtIndex:1] intValue]];
00196   NSNumber *twoAlpha = [NSNumber numberWithInt:[[twoComponents objectAtIndex:1] intValue]];
00197   return [oneAlpha compare:twoAlpha];
00198 }
