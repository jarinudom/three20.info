00054   NSCharacterSet* delimiterSet = [NSCharacterSet characterSetWithCharactersInString:@"&;"];
00055   NSMutableDictionary* pairs = [NSMutableDictionary dictionary];

00058     NSString* pairString = nil;
00059     [scanner scanUpToCharactersFromSet:delimiterSet intoString:&pairString];

00065       NSString* value = [[kvPair objectAtIndex:1]
00066                          stringByReplacingPercentEscapesUsingEncoding:encoding];

00068     }

00074 - (NSString*)stringByAddingQueryDictionary:(NSDictionary*)query {
00075   NSMutableArray* pairs = [NSMutableArray array];
00076   for (NSString* key in [query keyEnumerator]) {
00077     NSString* value = [query objectForKey:key];
00078     value = [value stringByReplacingOccurrencesOfString:@"?" withString:@"%3F"];
00079     value = [value stringByReplacingOccurrencesOfString:@"=" withString:@"%3D"];
00080     NSString* pair = [NSString stringWithFormat:@"%@=%@", key, value];
00081     [pairs addObject:pair];
00082   }
00083   
00084   NSString* params = [pairs componentsJoinedByString:@"&"];
00085   if ([self rangeOfString:@"?"].location == NSNotFound) {
00086     return [self stringByAppendingFormat:@"?%@", params];
00087   } else {
00088     return [self stringByAppendingFormat:@"&%@", params];
00089   }

00092 - (NSComparisonResult)versionStringCompare:(NSString *)other {
00093   NSArray *oneComponents = [self componentsSeparatedByString:@"a"];
00094   NSArray *twoComponents = [other componentsSeparatedByString:@"a"];
00095 
00096   // The parts before the "a"
00097   NSString *oneMain = [oneComponents objectAtIndex:0];
00098   NSString *twoMain = [twoComponents objectAtIndex:0];
00099 
00100   // If main parts are different, return that result, regardless of alpha part
00101   NSComparisonResult mainDiff;
00102   if ((mainDiff = [oneMain compare:twoMain]) != NSOrderedSame) {
00103     return mainDiff;
00104   }
00105 
00106   // At this point the main parts are the same; just deal with alpha stuff
00107   // If one has an alpha part and the other doesn't, the one without is newer
00108   if ([oneComponents count] < [twoComponents count]) {
00109     return NSOrderedDescending;
00110   } else if ([oneComponents count] > [twoComponents count]) {
00111     return NSOrderedAscending;
00112   } else if ([oneComponents count] == 1) {
00113     // Neither has an alpha part, and we know the main parts are the same
00114     return NSOrderedSame;
00115   }
00116 
00117   // At this point the main parts are the same and both have alpha parts. Compare the alpha parts
00118   // numerically. If it's not a valid number (including empty string) it's treated as zero.
00119   NSNumber *oneAlpha = [NSNumber numberWithInt:[[oneComponents objectAtIndex:1] intValue]];
00120   NSNumber *twoAlpha = [NSNumber numberWithInt:[[twoComponents objectAtIndex:1] intValue]];
00121   return [oneAlpha compare:twoAlpha];
00122 }
