00079 - (void)drawRect:(CGRect)rect {
00080   if (_numberOfPages > 1 || _hidesForSinglePage) {
00081     TTStyleContext* context = [[[TTStyleContext alloc] init] autorelease];
00082     TTBoxStyle* boxStyle = [self.normalDotStyle firstStyleOfClass:[TTBoxStyle class]];
00083 
00084     CGSize dotSize = [self.normalDotStyle addToSize:CGSizeZero context:context];
00085     
00086     CGFloat dotWidth = dotSize.width + boxStyle.margin.left + boxStyle.margin.right;
00087     CGFloat totalWidth = (dotWidth * _numberOfPages) - (boxStyle.margin.left + boxStyle.margin.right);
00088     CGRect contentRect = CGRectMake(round(self.width/2 - totalWidth/2),
00089                                     round(self.height/2 - dotSize.height/2),
00090                                     dotSize.width, dotSize.height);
00091       
00092     for (NSInteger i = 0; i < _numberOfPages; ++i) {
00093       contentRect.origin.x += boxStyle.margin.left;
00094 
00095       context.frame = contentRect;
00096       context.contentFrame = contentRect;
00097       
00098       if (i == _currentPage) {
00099         [self.currentDotStyle draw:context];
00100       } else {
00101         [self.normalDotStyle draw:context];
00102       }
00103       contentRect.origin.x += dotSize.width + boxStyle.margin.right;
00104     }
00105   }
00106 }

00109   TTStyleContext* context = [[[TTStyleContext alloc] init] autorelease];
00110   CGSize dotSize = [self.normalDotStyle addToSize:CGSizeZero context:context];
00111 
00112   CGFloat margin = 0;
00113   TTBoxStyle* boxStyle = [self.normalDotStyle firstStyleOfClass:[TTBoxStyle class]];
00114   if (boxStyle) {
00115     margin = boxStyle.margin.right + boxStyle.margin.left;

00118   return CGSizeMake((dotSize.width * _numberOfPages) + (margin * (_numberOfPages-1)),
