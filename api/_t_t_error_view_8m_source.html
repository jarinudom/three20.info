00083                       + kVPadding1 + kVPadding2;
00084   BOOL canShowImage = _imageView.image && self.height > maxHeight;
00085   
00086   CGFloat totalHeight = 0;
00087 
00088   if (canShowImage) {
00089     totalHeight += _imageView.height;
00090   }
00091   if (_titleView.text.length) {
00092     totalHeight += (totalHeight ? kVPadding1 : 0) + _titleView.height;

00094   if (_subtitleView.text.length) {
00095     totalHeight += (totalHeight ? kVPadding2 : 0) + _subtitleView.height;
00096   }
00097   
00098   CGFloat top = floor(self.height/2 - totalHeight/2);
00099   
00100   if (canShowImage) {
00101     _imageView.origin = CGPointMake(floor(self.width/2 - _imageView.width/2), top);
00102     _imageView.hidden = NO;
00103     top += _imageView.height + kVPadding1;
00104   } else {
00105     _imageView.hidden = YES;
00106   }
00107   if (_titleView.text.length) {
00108     _titleView.origin = CGPointMake(floor(self.width/2 - _titleView.width/2), top);
00109     top += _titleView.height + kVPadding2;
00110   }
00111   if (_subtitleView.text.length) {
00112     _subtitleView.origin = CGPointMake(floor(self.width/2 - _subtitleView.width/2), top);
