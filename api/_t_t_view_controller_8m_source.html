00037   NSValue* v1 = [notification.userInfo objectForKey:UIKeyboardBoundsUserInfoKey];
00038   CGRect keyboardBounds;

00042   CGPoint keyboardStart;
00043   [v2 getValue:&keyboardStart];
00044 
00045   NSValue* v3 = [notification.userInfo objectForKey:UIKeyboardCenterEndUserInfoKey];
00046   CGPoint keyboardEnd;
00047   [v3 getValue:&keyboardEnd];
00048 
00049   BOOL animated = keyboardStart.y != keyboardEnd.y;
00050   if (animated) {
00051     [UIView beginAnimations:nil context:nil];
00052     [UIView setAnimationDuration:TT_TRANSITION_DURATION];
00053   }
00054   
00055   if (appearing) {
00056     [self keyboardWillAppear:animated withBounds:keyboardBounds];
00057     
00058     [self retain];
00059     [NSTimer scheduledTimerWithTimeInterval:TT_TRANSITION_DURATION
00060              target:self selector:@selector(keyboardDidAppearDelayed:)
00061              userInfo:[NSValue valueWithCGRect:keyboardBounds] repeats:NO];
00062   } else {
00063     [self keyboardWillDisappear:animated withBounds:keyboardBounds];
00064 
00065     [self retain];
00066     [NSTimer scheduledTimerWithTimeInterval:TT_TRANSITION_DURATION
00067              target:self selector:@selector(keyboardDidDisappearDelayed:)
00068              userInfo:[NSValue valueWithCGRect:keyboardBounds] repeats:NO];
