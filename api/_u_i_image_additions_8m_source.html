00024 - (void)addRoundedRectToPath:(CGContextRef)context rect:(CGRect)rect radius:(float)radius {
00025   CGContextBeginPath(context);
00026   CGContextSaveGState(context);
00027 
00028   if (radius == 0) {
00029     CGContextTranslateCTM(context, CGRectGetMinX(rect), CGRectGetMinY(rect));
00030     CGContextAddRect(context, rect);
00031   } else {
00032     CGContextTranslateCTM(context, CGRectGetMinX(rect), CGRectGetMinY(rect));
00033     CGContextScaleCTM(context, radius, radius);
00034     float fw = CGRectGetWidth(rect) / radius;
00035     float fh = CGRectGetHeight(rect) / radius;
00036     
00037     CGContextMoveToPoint(context, fw, fh/2);
00038     CGContextAddArcToPoint(context, fw, fh, fw/2, fh, 1);
00039     CGContextAddArcToPoint(context, 0, fh, 0, fh/2, 1);
00040     CGContextAddArcToPoint(context, 0, 0, fw/2, 0, 1);
00041     CGContextAddArcToPoint(context, fw, 0, fw, fh/2, 1);
00042   }
00043 
00044   CGContextClosePath(context);
00045   CGContextRestoreGState(context);
00046 }

00051 - (UIImage*)transformWidth:(CGFloat)width height:(CGFloat)height rotate:(BOOL)rotate {
00052   CGFloat destW = width;
00053   CGFloat destH = height;
00054   CGFloat sourceW = width;
00055   CGFloat sourceH = height;
00056   if (rotate) {
00057     if (self.imageOrientation == UIImageOrientationRight
00058         || self.imageOrientation == UIImageOrientationLeft) {
00059       sourceW = height;
00060       sourceH = width;
00061     }
00062   }
00063   
00064   CGImageRef imageRef = self.CGImage;
00065   CGContextRef bitmap = CGBitmapContextCreate(NULL, destW, destH,
00066     CGImageGetBitsPerComponent(imageRef), CGImageGetBytesPerRow(imageRef), CGImageGetColorSpace(imageRef),
00067     CGImageGetBitmapInfo(imageRef));
00068 
00069   if (rotate) {
00070     if (self.imageOrientation == UIImageOrientationDown) {
00071       CGContextTranslateCTM(bitmap, sourceW, sourceH);
00072       CGContextRotateCTM(bitmap, 180 * (M_PI/180));
00073     } else if (self.imageOrientation == UIImageOrientationLeft) {
00074       CGContextTranslateCTM(bitmap, sourceH, 0);
00075       CGContextRotateCTM(bitmap, 90 * (M_PI/180));
00076     } else if (self.imageOrientation == UIImageOrientationRight) {
00077       CGContextTranslateCTM(bitmap, 0, sourceW);
00078       CGContextRotateCTM(bitmap, -90 * (M_PI/180));
00079     }
00080   }
00081 
00082   CGContextDrawImage(bitmap, CGRectMake(0,0,sourceW,sourceH), imageRef);
00083 
00084   CGImageRef ref = CGBitmapContextCreateImage(bitmap);
00085   UIImage* result = [UIImage imageWithCGImage:ref];
00086   CGContextRelease(bitmap);
00087   CGImageRelease(ref);
00088 
00089   return result;
