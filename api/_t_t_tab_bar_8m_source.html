00044 - (CGSize)layoutTabs {

00046   
00047   if (self.contentMode == UIViewContentModeScaleToFill) {
00048     CGFloat maxTextWidth = self.width - (kTabMargin*2 + kPadding*2*_tabViews.count);
00049     CGFloat totalTextWidth = 0;
00050     CGFloat totalTabWidth = kTabMargin*2;
00051     CGFloat maxTabWidth = 0;
00052     for (int i = 0; i < _tabViews.count; ++i) {
00053       TTTab* tab = [_tabViews objectAtIndex:i];
00054       [tab sizeToFit];
00055       totalTextWidth += tab.width - kPadding*2;
00056       totalTabWidth += tab.width;
00057       if (tab.width > maxTabWidth) {
00058         maxTabWidth = tab.width;
00059       }
00060     }
00061 
00062     if (totalTextWidth > maxTextWidth) {
00063       CGFloat shrinkFactor = maxTextWidth/totalTextWidth;
00064       for (int i = 0; i < _tabViews.count; ++i) {
00065         TTTab* tab = [_tabViews objectAtIndex:i];
00066         CGFloat textWidth = tab.width - kPadding*2;
00067         tab.frame = CGRectMake(x, 0, ceil(textWidth * shrinkFactor) + kPadding*2 , self.height);
00068         x += tab.width;
00069       }
00070     } else {
00071       CGFloat averageTabWidth = ceil((self.width - kTabMargin*2)/_tabViews.count);
00072       if (maxTabWidth > averageTabWidth && self.width - totalTabWidth < kTabMargin) {
00073         for (int i = 0; i < _tabViews.count; ++i) {
00074           TTTab* tab = [_tabViews objectAtIndex:i];
00075           tab.frame = CGRectMake(x, 0, tab.width, self.height);
00076           x += tab.width;
00077         }
00078       } else {
00079         for (int i = 0; i < _tabViews.count; ++i) {
00080           TTTab* tab = [_tabViews objectAtIndex:i];
00081           tab.frame = CGRectMake(x, 0, averageTabWidth, self.height);
00082           x += tab.width;
00083         }
00084       }
00085     }
00086   } else {
00087     for (int i = 0; i < _tabViews.count; ++i) {
00088       TTTab* tab = [_tabViews objectAtIndex:i];
