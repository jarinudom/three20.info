00047 - (CGSize)layoutTabs {
00048   CGFloat x = kTabMargin;
00049   
00050   if (self.contentMode == UIViewContentModeScaleToFill) {
00051     CGFloat maxTextWidth = self.width - (kTabMargin*2 + kPadding*2*_tabViews.count);
00052     CGFloat totalTextWidth = 0;
00053     CGFloat totalTabWidth = kTabMargin*2;
00054     CGFloat maxTabWidth = 0;
00055     for (int i = 0; i < _tabViews.count; ++i) {
00056       TTTab* tab = [_tabViews objectAtIndex:i];
00057       [tab sizeToFit];
00058       totalTextWidth += tab.width - kPadding*2;
00059       totalTabWidth += tab.width;
00060       if (tab.width > maxTabWidth) {
00061         maxTabWidth = tab.width;
00062       }
00063     }
00064 
00065     if (totalTextWidth > maxTextWidth) {
00066       CGFloat shrinkFactor = maxTextWidth/totalTextWidth;
00067       for (int i = 0; i < _tabViews.count; ++i) {
00068         TTTab* tab = [_tabViews objectAtIndex:i];
00069         CGFloat textWidth = tab.width - kPadding*2;
00070         tab.frame = CGRectMake(x, 0, ceil(textWidth * shrinkFactor) + kPadding*2 , self.height);
00071         x += tab.width;
00072       }
00073     } else {
00074       CGFloat averageTabWidth = ceil((self.width - kTabMargin*2)/_tabViews.count);
00075       if (maxTabWidth > averageTabWidth && self.width - totalTabWidth < kTabMargin) {
00076         for (int i = 0; i < _tabViews.count; ++i) {
00077           TTTab* tab = [_tabViews objectAtIndex:i];
00078           tab.frame = CGRectMake(x, 0, tab.width, self.height);
00079           x += tab.width;
00080         }
00081       } else {
00082         for (int i = 0; i < _tabViews.count; ++i) {
00083           TTTab* tab = [_tabViews objectAtIndex:i];
00084           tab.frame = CGRectMake(x, 0, averageTabWidth, self.height);
00085           x += tab.width;
00086         }
00087       }
00088     }
00089   } else {
00090     for (int i = 0; i < _tabViews.count; ++i) {
00091       TTTab* tab = [_tabViews objectAtIndex:i];
00092       [tab sizeToFit];
00093       tab.frame = CGRectMake(x, 0, tab.width, self.height);
00094       x += tab.width;
00095     }
00096   }
00097   
00098   return CGSizeMake(x, self.height);

00334 - (void)updateTabStyles {
00335   CGFloat columnCount = [self columnCount];
00336   int rowCount = [self rowCount];
00337   int cellCount = rowCount * columnCount;
00338 
00339   if (self.tabViews.count > columnCount) {
00340     int column = 0;
00341     for (TTTab* tab in self.tabViews) {
00342       if (column == 0) {
00343         [tab setStylesWithSelector:@"tabGridTabTopLeft:"];
00344       } else if (column == columnCount-1) {
00345         [tab setStylesWithSelector:@"tabGridTabTopRight:"];
00346       } else if (column == cellCount - columnCount) {
00347         [tab setStylesWithSelector:@"tabGridTabBottomLeft:"];
00348       } else if (column == cellCount - 1) {
00349         [tab setStylesWithSelector:@"tabGridTabBottomRight:"];
