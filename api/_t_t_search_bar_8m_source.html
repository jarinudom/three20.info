00063     if (indexView) {

00067       if (show) {

00070         indexView.frame = frame;

00072         indexView.frame = CGRectOffset(indexView.frame, indexView.width + kIndexViewMargin, 0);
00073       }
00074       indexView.alpha = show ? 1 : 0;
00075       
00076       CGRect searchFrame = _searchField.frame;
00077       searchFrame.size.width += show ? -self.indexViewWidth : self.indexViewWidth;
00078       _searchField.frame = searchFrame;

00081       boxFrame.size.width += show ? -self.indexViewWidth : self.indexViewWidth;
00082       _boxView.frame = boxFrame;

00160 - (void)layoutSubviews {
00161   CGFloat indexViewWidth = [_searchField isEditing] ? 0 : self.indexViewWidth;
00162   CGFloat leftPadding = _showsSearchIcon ? 0 : kSpacingX;
00163 
00164   CGFloat buttonWidth = 0;
00165   if (_showsCancelButton) {
00166     [_cancelButton sizeToFit];
00167     buttonWidth = _cancelButton.width + kButtonSpacing;
00168   }
00169 
00170   CGFloat boxHeight = self.font.ttLineHeight + 8;
00171   _boxView.frame = CGRectMake(kMarginX, floor(self.height/2 - boxHeight/2),
00172                               self.width - (kMarginX*2 + indexViewWidth + buttonWidth), boxHeight);
00173     
00174   _searchField.frame = CGRectMake(kMarginX+kPaddingX+leftPadding, 0,
00175     self.width - (kMarginX*2+kPaddingX+leftPadding+buttonWidth+indexViewWidth), self.height);
00176   
00177   if (_showsCancelButton) {
00178     _cancelButton.frame = CGRectMake(_boxView.right + kButtonSpacing,
00179                                      floor(self.height/2 - kButtonHeight/2),
00180                                      _cancelButton.width, kButtonHeight);
00181   }
00182 }
