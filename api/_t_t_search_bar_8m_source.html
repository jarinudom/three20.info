00064       if (show) {
00065         CGRect frame = indexView.frame;
00066         frame.origin.x = self.width - (indexView.width + kIndexViewMargin);
00067         indexView.frame = frame;

00069         indexView.frame = CGRectOffset(indexView.frame, indexView.width + kIndexViewMargin, 0);
00070       }
00071       indexView.alpha = show ? 1 : 0;
00072       
00073       CGRect searchFrame = _searchField.frame;
00074       searchFrame.size.width += show ? -self.indexViewWidth : self.indexViewWidth;
00075       _searchField.frame = searchFrame;

00077       CGRect boxFrame = _boxView.frame;
00078       boxFrame.size.width += show ? -self.indexViewWidth : self.indexViewWidth;
00079       _boxView.frame = boxFrame;

00157 - (void)layoutSubviews {
00158   CGFloat indexViewWidth = [_searchField isEditing] ? 0 : self.indexViewWidth;
00159   CGFloat leftPadding = _showsSearchIcon ? 0 : kSpacingX;
00160 
00161   CGFloat buttonWidth = 0;
00162   if (_showsCancelButton) {
00163     [_cancelButton sizeToFit];
00164     buttonWidth = _cancelButton.width + kButtonSpacing;
00165   }
00166 
00167   CGFloat boxHeight = self.font.lineHeight + 8;
00168   _boxView.frame = CGRectMake(kMarginX, floor(self.height/2 - boxHeight/2),
00169                               self.width - (kMarginX*2 + indexViewWidth + buttonWidth), boxHeight);
00170     
00171   _searchField.frame = CGRectMake(kMarginX+kPaddingX+leftPadding, 0,
00172     self.width - (kMarginX*2+kPaddingX+leftPadding+buttonWidth+indexViewWidth), self.height);
00173   
00174   if (_showsCancelButton) {
00175     _cancelButton.frame = CGRectMake(_boxView.right + kButtonSpacing,
00176                                      floor(self.height/2 - kButtonHeight/2),
00177                                      _cancelButton.width, kButtonHeight);
00178   }
00179 }
