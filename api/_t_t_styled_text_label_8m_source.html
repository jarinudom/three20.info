00128 - (void)addAccessibilityElementsForNode:(TTStyledNode*)node {
00129   if ([node isKindOfClass:[TTStyledLinkNode class]]) {
00130     UIAccessibilityElement* acc = [[[UIAccessibilityElement alloc]
00131                                   initWithAccessibilityContainer:self] autorelease];
00132     TTStyledFrame* frame = [_text getFrameForNode:node];
00133     acc.accessibilityFrame = CGRectOffset(frame.bounds, self.screenViewX, self.screenViewY);
00134     acc.accessibilityTraits = UIAccessibilityTraitLink;
00135     acc.accessibilityLabel = [node outerText];
00136     [_accessibilityElements addObject:acc];
00137   } else if ([node isKindOfClass:[TTStyledTextNode class]]) {
00138     TTStyledTextFrame* startFrame = (TTStyledTextFrame*)[_text getFrameForNode:node];
00139     UIEdgeInsets edges = [self edgesForRect:startFrame.bounds];
00140   
00141     TTStyledTextFrame* frame = (TTStyledTextFrame*)startFrame.nextFrame;
00142     for (; [frame isKindOfClass:[TTStyledTextFrame class]]; frame = (TTStyledTextFrame*)frame.nextFrame) {
00143       if (frame.bounds.origin.x < edges.left) {
00144         [self addAccessibilityElementFromFrame:startFrame toFrame:frame withEdges:edges];
00145         edges = [self edgesForRect:frame.bounds];
00146         startFrame = frame;
00147       } else {
00148         if (frame.bounds.origin.x+frame.bounds.size.width > edges.right) {
00149           edges.right = frame.bounds.origin.x+frame.bounds.size.width;
00150         }
00151         if (frame.bounds.origin.y+frame.bounds.size.height > edges.bottom) {
00152           edges.bottom = frame.bounds.origin.y+frame.bounds.size.height;
00153         }
00154       }
00155     }
00156     
00157     if (frame != startFrame) {
00158       [self addAccessibilityElementFromFrame:startFrame toFrame:frame withEdges:edges];
00159     }
00160   } else if ([node isKindOfClass:[TTStyledElement class]]) {
00161     TTStyledElement* element = (TTStyledElement*)node;
00162     for (TTStyledNode* child = element.firstChild; child; child = child.nextSibling) {
00163       [self addAccessibilityElementsForNode:child];
00164     }
00165   }
00166 }
