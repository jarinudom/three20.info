00022 #define MAX3(a,b,c) (a > b ? (a > c ? a : c) : (b > c ? b : c))
00023 #define MIN3(a,b,c) (a < b ? (a < c ? a : c) : (b < c ? b : c))

00025 void RGBtoHSV(float r, float g, float b, float* h, float* s, float* v) {
00026   float min, max, delta;
00027   min = MIN3(r, g, b);
00028   max = MAX3(r, g, b);
00029   *v = max;       // v
00030   delta = max - min;
00031   if( max != 0 )
00032     *s = delta / max;   // s
00033   else {
00034     // r = g = b = 0    // s = 0, v is undefined
00035     *s = 0;
00036     *h = -1;
00037     return;
00038   }
00039   if( r == max )
00040     *h = ( g - b ) / delta;   // between yellow & magenta
00041   else if( g == max )
00042     *h = 2 + ( b - r ) / delta; // between cyan & yellow
00043   else
00044     *h = 4 + ( r - g ) / delta; // between magenta & cyan
00045   *h *= 60;       // degrees
00046   if( *h < 0 )
00047     *h += 360;
00048 }

00050 void HSVtoRGB( float *r, float *g, float *b, float h, float s, float v )
00051 {
00052   int i;
00053   float f, p, q, t;
00054   if( s == 0 ) {
00055     // achromatic (grey)
00056     *r = *g = *b = v;
00057     return;
00058   }
00059   h /= 60;      // sector 0 to 5
00060   i = floor( h );
00061   f = h - i;      // factorial part of h
00062   p = v * ( 1 - s );
00063   q = v * ( 1 - s * f );
00064   t = v * ( 1 - s * ( 1 - f ) );
00065   switch( i ) {
00066     case 0:
00067       *r = v;
00068       *g = t;
00069       *b = p;
00070       break;
00071     case 1:
00072       *r = q;
00073       *g = v;
00074       *b = p;
00075       break;
00076     case 2:
00077       *r = p;
00078       *g = v;
00079       *b = t;
00080       break;
00081     case 3:
00082       *r = p;
00083       *g = q;
00084       *b = v;
00085       break;
00086     case 4:
00087       *r = t;
00088       *g = p;
00089       *b = v;
00090       break;
00091     default:    // case 5:
00092       *r = v;
00093       *g = p;
00094       *b = q;
00095       break;
00096   }
00097 }
