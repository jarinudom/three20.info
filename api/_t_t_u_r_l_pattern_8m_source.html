00695     return NO;
00696   }
00697 
00698   NSArray* pathComponents = URL.path.pathComponents;
00699   NSInteger componentCount = URL.path.length ? pathComponents.count : (URL.host ? 1 : 0);
00700   if (componentCount != _path.count) {
00701     return NO;
00702   }
00703 
00704   if (_path.count && URL.host) {
00705     id<TTURLPatternText>hostPattern = [_path objectAtIndex:0];
00706     if (![hostPattern match:URL.host]) {
00707       return NO;
00708     }
00709   }
00710   
00711   for (NSInteger i = 1; i < _path.count; ++i) {
00712     id<TTURLPatternText>pathPattern = [_path objectAtIndex:i];
00713     NSString* pathText = [pathComponents objectAtIndex:i];
00714     if (![pathPattern match:pathText]) {
00715       return NO;
00716     }
00717   }
00718   
00719   if ((URL.fragment && !_fragment) || (_fragment && !URL.fragment)) {
00720     return NO;
