00690 - (BOOL)matchURL:(NSURL*)URL {
00691   if (!URL.scheme || !URL.host || ![_scheme isEqualToString:URL.scheme]) {
00692     return NO;
00693   }
00694 
00695   NSArray* pathComponents = URL.path.pathComponents;
00696   NSInteger componentCount = URL.path.length ? pathComponents.count : (URL.host ? 1 : 0);
00697   if (componentCount != _path.count) {
00698     return NO;
00699   }
00700 
00701   if (_path.count && URL.host) {
00702     id<TTURLPatternText>hostPattern = [_path objectAtIndex:0];
00703     if (![hostPattern match:URL.host]) {
00704       return NO;
00705     }
00706   }
00707   
00708   for (NSInteger i = 1; i < _path.count; ++i) {
00709     id<TTURLPatternText>pathPattern = [_path objectAtIndex:i];
00710     NSString* pathText = [pathComponents objectAtIndex:i];
00711     if (![pathPattern match:pathText]) {
00712       return NO;
00713     }
00714   }
00715   
00716   if ((URL.fragment && !_fragment) || (_fragment && !URL.fragment)) {
00717     return NO;
00718   } else if (URL.fragment && _fragment && ![_fragment match:URL.fragment]) {
00719     return NO;
00720   }
