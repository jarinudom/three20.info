00133 - (void)layoutSubviews {
00134   CGRect screenBounds = TTScreenBounds();
00135   CGFloat width = self.width;
00136   CGFloat height = self.height;
00137   CGFloat cx = self.bounds.origin.x + width/2;
00138   CGFloat cy = self.bounds.origin.y + height/2;
00139   CGFloat marginRight = 0, marginLeft = 0, marginBottom = TTToolbarHeight();
00140 
00141   // Since the photo view is constrained to the size of the image, but we want to position
00142   // the status views relative to the screen, offset by the difference
00143   CGFloat screenOffset = -floor(screenBounds.size.height/2 - height/2);
00144   
00145   // Vertically center in the space between the bottom of the image and the bottom of the screen
00146   CGFloat imageBottom = screenBounds.size.height/2 + self.defaultImage.size.height/2;
00147   CGFloat textWidth = screenBounds.size.width - (marginLeft+marginRight);
00148   
00149   if (_statusLabel.text.length) {
00150     CGSize statusSize = [_statusLabel sizeThatFits:CGSizeMake(textWidth, 0)];
00151     _statusLabel.frame = 
00152         CGRectMake(marginLeft + (cx - screenBounds.size.width/2), 
00153                    cy + floor(screenBounds.size.height/2 - (statusSize.height+marginBottom)),
00154                    textWidth, statusSize.height);
00155   } else {
00156     _statusLabel.frame = CGRectZero;
00157   }
00158   
00159   if (_captionLabel.text.length) {
00160     CGSize captionSize = [_captionLabel sizeThatFits:CGSizeMake(textWidth, 0)];
00161     _captionLabel.frame = CGRectMake(marginLeft + (cx - screenBounds.size.width/2), 
00162                                      cy + floor(screenBounds.size.height/2
00163                                                 - (captionSize.height+marginBottom)),
00164                                      textWidth, captionSize.height);
00165   } else {
00166     _captionLabel.frame = CGRectZero;
00167   }
00168   
00169   CGFloat spinnerTop = _captionLabel.height
00170     ? _captionLabel.top - floor(_statusSpinner.height + _statusSpinner.height/2)
00171     : screenOffset + imageBottom + floor(_statusSpinner.height/2);
00172 
00173   _statusSpinner.frame =
00174     CGRectMake(self.bounds.origin.x + floor(self.bounds.size.width/2 - _statusSpinner.width/2),
00175                spinnerTop, _statusSpinner.width, _statusSpinner.height);
