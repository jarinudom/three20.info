00137 - (void)layoutSubviews {
00138   CGRect screenBounds = TTScreenBounds();
00139   CGFloat width = self.width;
00140   CGFloat height = self.height;
00141   CGFloat cx = self.bounds.origin.x + width/2;
00142   CGFloat cy = self.bounds.origin.y + height/2;
00143   CGFloat marginRight = 0, marginLeft = 0, marginBottom = TTToolbarHeight();
00144 
00145   // Since the photo view is constrained to the size of the image, but we want to position
00146   // the status views relative to the screen, offset by the difference
00147   CGFloat screenOffset = -floor(screenBounds.size.height/2 - height/2);
00148   
00149   // Vertically center in the space between the bottom of the image and the bottom of the screen
00150   CGFloat imageBottom = screenBounds.size.height/2 + self.defaultImage.size.height/2;
00151   CGFloat textWidth = screenBounds.size.width - (marginLeft+marginRight);
00152   
00153   if (_statusLabel.text.length) {
00154     CGSize statusSize = [_statusLabel sizeThatFits:CGSizeMake(textWidth, 0)];
00155     _statusLabel.frame = 
00156         CGRectMake(marginLeft + (cx - screenBounds.size.width/2), 
00157                    cy + floor(screenBounds.size.height/2 - (statusSize.height+marginBottom)),
00158                    textWidth, statusSize.height);
00159   } else {
00160     _statusLabel.frame = CGRectZero;
00161   }
00162   
00163   if (_captionLabel.text.length) {
00164     CGSize captionSize = [_captionLabel sizeThatFits:CGSizeMake(textWidth, 0)];
00165     _captionLabel.frame = CGRectMake(marginLeft + (cx - screenBounds.size.width/2), 
00166                                      cy + floor(screenBounds.size.height/2
00167                                                 - (captionSize.height+marginBottom)),
00168                                      textWidth, captionSize.height);
00169   } else {
00170     _captionLabel.frame = CGRectZero;
00171   }
00172   
00173   CGFloat spinnerTop = _captionLabel.height
00174     ? _captionLabel.top - floor(_statusSpinner.height + _statusSpinner.height/2)
00175     : screenOffset + imageBottom + floor(_statusSpinner.height/2);
00176 
00177   _statusSpinner.frame =
00178     CGRectMake(self.bounds.origin.x + floor(self.bounds.size.width/2 - _statusSpinner.width/2),
00179                spinnerTop, _statusSpinner.width, _statusSpinner.height);
00180 
00181 }
