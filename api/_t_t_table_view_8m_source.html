00109     // As described below in setContentOffset, UITableView insists on messing with the

00115 - (void)setContentOffset:(CGPoint)point {
00116   // UITableView (and UIScrollView) are really stupid about resetting the content offset
00117   // when the table view itself is resized.  There are times when I scroll to a point and then
00118   // disable scrolling, and I don't want the table view scrolling somewhere else just because
00119   // it was resized.  
00120   if (self.scrollEnabled) {
00121     if (!(_contentOrigin && self.contentOffset.y == _contentOrigin && point.y == 0)) {
00122       [super setContentOffset:point];
00123     }
00124   }
00125 }
