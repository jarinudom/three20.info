00226 - (void)parseText:(NSString*)string {
00227   if (_parseLineBreaks) {
00228     NSCharacterSet* newLines = [NSCharacterSet newlineCharacterSet];
00229     NSInteger index = 0;
00230     NSInteger length = string.length;
00231     while (1) {
00232       NSRange searchRange = NSMakeRange(index, length - index);
00233       NSRange range = [string rangeOfCharacterFromSet:newLines options:0 range:searchRange];
00234       if (range.location != NSNotFound) {
00235         // Find all text before the line break and parse it
00236         NSRange textRange = NSMakeRange(index, range.location - index);
00237         NSString* substr = [string substringWithRange:textRange];
00238         [self parseURLs:substr];
00239         
00240         // Add a line break node after the text
00241         TTStyledLineBreakNode* br = [[[TTStyledLineBreakNode alloc] init] autorelease];
00242         [self addNode:br];
00243 
00244         index = index + substr.length + 1;
00245       } else {
00246         // Find all text until the end of hte string and parse it
00247         NSString* substr = [string substringFromIndex:index];
00248         [self parseURLs:substr];
00249         break;
00250       }
00251     }
00252   } else if (_parseURLs) {
00253     [self parseURLs:string];
