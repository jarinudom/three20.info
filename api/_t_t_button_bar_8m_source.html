00066 - (void)layoutSubviews {
00067   CGFloat buttonWidth = floor((self.width-kPadding) / _buttons.count)-kPadding;
00068   if (buttonWidth > kButtonMaxWidth) {
00069     buttonWidth = kButtonMaxWidth;
00070   }
00071 
00072   CGFloat x = kPadding + floor(self.width/2 - ((buttonWidth+kPadding)*_buttons.count)/2);
00073   CGFloat y = floor(self.height/2 - kButtonHeight/2);
00074   
00075   for (UIButton* button in _buttons) {
00076     button.frame = CGRectMake(x, y, buttonWidth, kButtonHeight);
00077     x += button.width + kPadding;
00078   }
00079 }
