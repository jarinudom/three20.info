00063 - (void)layoutSubviews {
00064   CGFloat buttonWidth = floor((self.width-kPadding) / _buttons.count)-kPadding;
00065   if (buttonWidth > kButtonMaxWidth) {
00066     buttonWidth = kButtonMaxWidth;
00067   }
00068 
00069   CGFloat x = kPadding + floor(self.width/2 - ((buttonWidth+kPadding)*_buttons.count)/2);
00070   CGFloat y = floor(self.height/2 - kButtonHeight/2);
00071   
00072   for (UIButton* button in _buttons) {
00073     button.frame = CGRectMake(x, y, buttonWidth, kButtonHeight);
00074     x += button.width + kPadding;
00075   }
00076 }
