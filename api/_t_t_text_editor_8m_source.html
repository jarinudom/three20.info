00051 - (void)setContentOffset:(CGPoint)offset animated:(BOOL)animated {
00052   if (_autoresizesToText) {
00053     if (!_overflowed) {
00054       // In autosizing mode, we don't ever allow the text view to scroll past zero
00055       // unless it has past its maximum number of lines
00056       [super setContentOffset:CGPointZero animated:animated];
00057     } else {
00058       // If there is an overflow, we force the text view to keep the cursor at the bottom of the
00059       // view.
00060       [super setContentOffset: CGPointMake(offset.x, self.contentSize.height - self.height)
00061                      animated: animated];
00062     }
00063   } else {
00064     [super setContentOffset:offset animated:animated];
00065   }
00066 }

00290 - (CGFloat)heightThatFits:(BOOL*)overflowed numberOfLines:(NSInteger*)numberOfLines {
00291   CGFloat ttLineHeight = self.font.ttLineHeight;
00292   CGFloat minHeight = _minNumberOfLines * ttLineHeight;
00293   CGFloat maxHeight = _maxNumberOfLines * ttLineHeight;
00294   CGFloat maxWidth = self.width - kTextViewInset;
00295   
00296   NSString* text = _textField.hidden ? _textView.text : _textField.text;
00297   if (!text.length) {
00298     text = @"M";
00299   }
00300 
00301   CGSize textSize = [text sizeWithFont:self.font
00302                           constrainedToSize:CGSizeMake(maxWidth, CGFLOAT_MAX)
00303                           lineBreakMode:UILineBreakModeWordWrap];
00304   
00305   CGFloat newHeight = textSize.height;
00306   if ([text characterAtIndex:text.length-1] == 10) {
00307     newHeight += ttLineHeight;
00308   }
00309   if (_showsExtraLine) {
00310     newHeight += ttLineHeight;
00311   }
00312 
00313   if (overflowed) {
00314     *overflowed = maxHeight && newHeight > maxHeight;
00315   }
00316 
00317   if (numberOfLines) {
00318     *numberOfLines = floor(newHeight / ttLineHeight);
00319   }
00320   
00321   if (newHeight < minHeight) {
00322     newHeight = minHeight;
00323   }
00324   if (maxHeight && newHeight > maxHeight) {
00325     newHeight = maxHeight;
00326   }
00327   
00328   return newHeight + kPaddingY*2;
00329 }

00335 - (void)constrainToText {
00336   NSInteger numberOfLines = 0;
00337   CGFloat oldHeight = self.height;
00338   CGFloat newHeight = [self heightThatFits:&_overflowed numberOfLines:&numberOfLines];
00339   CGFloat diff = newHeight - oldHeight;
00340     
00341   if (numberOfLines > 1 && !_textField.hidden) {
00342     [self createTextView];
00343     _textField.hidden = YES;
00344     _textView.hidden = NO;
00345     _textView.text = _textField.text;
00346     _internal.ignoreBeginAndEnd = YES;
00347     [_textView becomeFirstResponder];
00348     [self performSelector:@selector(stopIgnoringBeginAndEnd) withObject:nil afterDelay:0];
00349   } else if (numberOfLines == 1 && _textField.hidden) {
00350     _textField.hidden = NO;
00351     _textView.hidden = YES;
00352     _textField.text = _textView.text;
00353     _internal.ignoreBeginAndEnd = YES;
00354     [_textField becomeFirstResponder];
00355     [self performSelector:@selector(stopIgnoringBeginAndEnd) withObject:nil afterDelay:0];
00356   }
00357   
00358   _textView.overflowed = _overflowed;
00359   _textView.scrollEnabled = _overflowed;
00360   
00361   if (oldHeight && diff) {
00362     if ([_delegate respondsToSelector:@selector(textEditor:shouldResizeBy:)]) {
00363       if (![_delegate textEditor:self shouldResizeBy:diff]) {
00364         return;
00365       }
00366     }
00367     
00368     self.frame = TTRectContract(self.frame, 0, -diff);
00369   }
00370 }
