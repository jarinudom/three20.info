00049 - (void)setContentOffset:(CGPoint)offset animated:(BOOL)animated {
00050   if (_autoresizesToText) {
00051     if (!_overflowed) {
00052       // In autosizing mode, we don't ever allow the text view to scroll past zero
00053       // unless it has past its maximum number of lines
00054       [super setContentOffset:CGPointZero animated:animated];
00055     } else {
00056       // If there is an overflow, we force the text view to keep the cursor at the bottom of the
00057       // view.
00058       [super setContentOffset: CGPointMake(offset.x, self.contentSize.height - self.height)
00059                      animated: animated];
00060     }
00061   } else {
00062     [super setContentOffset:offset animated:animated];
00063   }

00288 - (CGFloat)heightThatFits:(BOOL*)overflowed numberOfLines:(NSInteger*)numberOfLines {
00289   CGFloat lineHeight = self.font.lineHeight;
00290   CGFloat minHeight = _minNumberOfLines * lineHeight;
00291   CGFloat maxHeight = _maxNumberOfLines * lineHeight;
00292   CGFloat maxWidth = self.width - kTextViewInset;
00293   
00294   NSString* text = _textField.hidden ? _textView.text : _textField.text;
00295   if (!text.length) {
00296     text = @"M";
00297   }
00298 
00299   CGSize textSize = [text sizeWithFont:self.font
00300                           constrainedToSize:CGSizeMake(maxWidth, CGFLOAT_MAX)
00301                           lineBreakMode:UILineBreakModeWordWrap];
00302   
00303   CGFloat newHeight = textSize.height;
00304   if ([text characterAtIndex:text.length-1] == 10) {
00305     newHeight += lineHeight;
00306   }
00307   if (_showsExtraLine) {
00308     newHeight += lineHeight;
00309   }
00310 
00311   if (overflowed) {
00312     *overflowed = maxHeight && newHeight > maxHeight;
00313   }
00314 
00315   if (numberOfLines) {
00316     *numberOfLines = floor(newHeight / lineHeight);
00317   }
00318   
00319   if (newHeight < minHeight) {
00320     newHeight = minHeight;
00321   }
00322   if (maxHeight && newHeight > maxHeight) {
00323     newHeight = maxHeight;
00324   }
00325   
00326   return newHeight + kPaddingY*2;
00327 }

00333 - (void)constrainToText {
00334   NSInteger numberOfLines = 0;
00335   CGFloat oldHeight = self.height;
00336   CGFloat newHeight = [self heightThatFits:&_overflowed numberOfLines:&numberOfLines];
00337   CGFloat diff = newHeight - oldHeight;
00338     
00339   if (numberOfLines > 1 && !_textField.hidden) {
00340     [self createTextView];
00341     _textField.hidden = YES;
00342     _textView.hidden = NO;
00343     _textView.text = _textField.text;
00344     _internal.ignoreBeginAndEnd = YES;
00345     [_textView becomeFirstResponder];
00346     [self performSelector:@selector(stopIgnoringBeginAndEnd) withObject:nil afterDelay:0];
00347   } else if (numberOfLines == 1 && _textField.hidden) {
00348     _textField.hidden = NO;
00349     _textView.hidden = YES;
00350     _textField.text = _textView.text;
00351     _internal.ignoreBeginAndEnd = YES;
00352     [_textField becomeFirstResponder];
00353     [self performSelector:@selector(stopIgnoringBeginAndEnd) withObject:nil afterDelay:0];
00354   }
00355   
00356   _textView.overflowed = _overflowed;
00357   _textView.scrollEnabled = _overflowed;
00358   
00359   if (oldHeight && diff) {
00360     if ([_delegate respondsToSelector:@selector(textEditor:shouldResizeBy:)]) {
00361       if (![_delegate textEditor:self shouldResizeBy:diff]) {
00362         return;
00363       }
00364     }
00365     
00366     self.frame = TTRectContract(self.frame, 0, -diff);
00367   }
00368 }
