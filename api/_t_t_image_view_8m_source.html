00217 - (void)setImage:(UIImage*)image {
00218   if (image != _image) {
00219     [_image release];
00220     _image = [image retain];
00221 
00222     [self updateLayer];
00223     CGRect frame = self.frame;
00224     if (_autoresizesToImage) {
00225       self.frame = CGRectMake(frame.origin.x, frame.origin.y, image.size.width, image.size.height);
00226     } else {
00227       if (!frame.size.width && !frame.size.height) {
00228         self.frame = CGRectMake(frame.origin.x, frame.origin.y, image.size.width, image.size.height);
00229       } else if (frame.size.width && !frame.size.height) {
00230         self.frame = CGRectMake(frame.origin.x, frame.origin.y,
00231           frame.size.width, floor((image.size.height/image.size.width) * frame.size.width));
00232       } else if (frame.size.height && !frame.size.width) {
00233         self.frame = CGRectMake(frame.origin.x, frame.origin.y,
00234           floor((image.size.width/image.size.height) * frame.size.height), frame.size.height);
00235       }
00236     }
00237 
00238     if (!_defaultImage || image != _defaultImage) {
00239       [self imageViewDidLoadImage:image];
00240       if ([_delegate respondsToSelector:@selector(imageView:didLoadImage:)]) {
00241         [_delegate imageView:self didLoadImage:image];
