00212 - (void)setImage:(UIImage*)image {
00213   if (image != _image) {
00214     [_image release];
00215     _image = [image retain];
00216 
00217     [self updateLayer];
00218     CGRect frame = self.frame;
00219     if (_autoresizesToImage) {
00220       self.frame = CGRectMake(frame.origin.x, frame.origin.y, image.size.width, image.size.height);
00221     } else {
00222       if (!frame.size.width && !frame.size.height) {
00223         self.frame = CGRectMake(frame.origin.x, frame.origin.y, image.size.width, image.size.height);
00224       } else if (frame.size.width && !frame.size.height) {
00225         self.frame = CGRectMake(frame.origin.x, frame.origin.y,
00226           frame.size.width, floor((image.size.height/image.size.width) * frame.size.width));
00227       } else if (frame.size.height && !frame.size.width) {
00228         self.frame = CGRectMake(frame.origin.x, frame.origin.y,
00229           floor((image.size.width/image.size.height) * frame.size.height), frame.size.height);
00230       }
00231     }
00232 
00233     if (!_defaultImage || image != _defaultImage) {
00234       [self imageViewDidLoadImage:image];
00235       if ([_delegate respondsToSelector:@selector(imageView:didLoadImage:)]) {
00236         [_delegate imageView:self didLoadImage:image];
