00057   CGFloat x = _padding, y = _padding;

00060   for (UIView* subview in subviews) {
00061     if (x + subview.width > maxWidth) {
00062       x = _padding;

00104 - (CGSize)layoutSubviews:(NSArray*)subviews forView:(UIView*)view {
00105   CGFloat innerWidth = (view.width - _padding*2);
00106   CGFloat width = ceil(innerWidth / _columnCount);
00107   CGFloat rowHeight = 0;
00108   
00109   CGFloat x = _padding, y = _padding;
00110   CGFloat maxX = 0, lastHeight = 0;
00111   NSInteger column = 0;
00112   for (UIView* subview in subviews) {
00113     if (column % _columnCount == 0) {
00114       x = _padding;
00115       y += rowHeight + _spacing;
00116     }
00117     CGSize size = [subview sizeThatFits:CGSizeMake(width, 0)];
00118     rowHeight = size.height;
00119     subview.frame = CGRectMake(x, y, width, size.height);
