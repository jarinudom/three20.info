00064 - (CGFloat)layoutCells {
00065   CGFloat fontHeight = self.font.lineHeight;
00066   CGFloat lineIncrement = fontHeight + kCellPaddingY*2 + kSpacingY;
00067   CGFloat marginY = floor(fontHeight/kPaddingRatio);
00068   CGFloat marginLeft = self.leftView
00069     ? kPaddingX + self.leftView.width + kPaddingX/2
00070     : kPaddingX;
00071   CGFloat marginRight = kPaddingX + (self.rightView ? kClearButtonSize : 0);
00072 
00073   _cursorOrigin.x = marginLeft;
00074   _cursorOrigin.y = marginY;
00075   _lineCount = 1;
00076 
00077   if (self.width) {
00078     for (TTPickerViewCell* cell in _cellViews) {
00079       [cell sizeToFit];
00080 
00081       CGFloat lineWidth = _cursorOrigin.x + cell.frame.size.width + marginRight;
00082       if (lineWidth >= self.width) {
00083         _cursorOrigin.x = marginLeft;
00084         _cursorOrigin.y += lineIncrement;
00085         ++_lineCount;
00086       }
00087 
00088       cell.frame = CGRectMake(_cursorOrigin.x, _cursorOrigin.y-kCellPaddingY,
00089         cell.width, cell.height);
00090       _cursorOrigin.x += cell.frame.size.width + kPaddingX;
00091     }
00092 
00093     CGFloat remainingWidth = self.width - (_cursorOrigin.x + marginRight);
00094     if (remainingWidth < kMinCursorWidth) {
00095       _cursorOrigin.x = marginLeft;
00096       _cursorOrigin.y += lineIncrement;
00097         ++_lineCount;
00098     }
00099   }
00100   
00101   return _cursorOrigin.y + fontHeight + marginY;
00102 }
