00070 - (CGFloat)layoutCells {
00071   CGFloat fontHeight = self.font.ttLineHeight;
00072   CGFloat lineIncrement = fontHeight + kCellPaddingY*2 + kSpacingY;
00073   CGFloat marginY = floor(fontHeight/kPaddingRatio);
00074   CGFloat marginLeft = self.leftView
00075     ? kPaddingX + self.leftView.width + kPaddingX/2
00076     : kPaddingX;
00077   CGFloat marginRight = kPaddingX + (self.rightView ? kClearButtonSize : 0);
00078 
00079   _cursorOrigin.x = marginLeft;
00080   _cursorOrigin.y = marginY;
00081   _lineCount = 1;
00082 
00083   if (self.width) {
00084     for (TTPickerViewCell* cell in _cellViews) {
00085       [cell sizeToFit];
00086 
00087       CGFloat lineWidth = _cursorOrigin.x + cell.frame.size.width + marginRight;
00088       if (lineWidth >= self.width) {
00089         _cursorOrigin.x = marginLeft;
00090         _cursorOrigin.y += lineIncrement;
00091         ++_lineCount;
00092       }
00093 
00094       cell.frame = CGRectMake(_cursorOrigin.x, _cursorOrigin.y-kCellPaddingY,
00095         cell.width, cell.height);
00096       _cursorOrigin.x += cell.frame.size.width + kPaddingX;
00097     }
00098 
00099     CGFloat remainingWidth = self.width - (_cursorOrigin.x + marginRight);
00100     if (remainingWidth < kMinCursorWidth) {
00101       _cursorOrigin.x = marginLeft;
00102       _cursorOrigin.y += lineIncrement;
00103         ++_lineCount;
00104     }
00105   }
00106   
00107   return _cursorOrigin.y + fontHeight + marginY;
00108 }
