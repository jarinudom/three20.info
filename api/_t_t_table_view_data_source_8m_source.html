00109             atIndex:(NSInteger)index {

00111     if (index == 0)  {
00112       // This is a hack to get the table header to appear when the user touches the

00121   NSInteger sectionCount = [tableView numberOfSections];
00122   for (NSInteger i = 0; i < sectionCount; ++i) {
00123     NSString* section  = [tableView.dataSource tableView:tableView titleForHeaderInSection:i];
00124     if ([section hasPrefix:letter]) {
00125       return i;
00126     }
00127   }
00128   if (index >= sectionCount) {
00129     return sectionCount-1;
