00117   NSInteger sectionCount = [tableView numberOfSections];
00118   for (NSInteger i = 0; i < sectionCount; ++i) {
00119     NSString* section  = [tableView.dataSource tableView:tableView titleForHeaderInSection:i];

00121       return i;
00122     }
00123   }
00124   if (index >= sectionCount) {
00125     return sectionCount-1;
