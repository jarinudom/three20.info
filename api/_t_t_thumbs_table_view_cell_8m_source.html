00123     if (columnCount > _columnCount) {
00124       for (TTThumbView* thumbView in _thumbViews) {

00129     
00130     _columnCount = columnCount;
00131     
00132     for (NSInteger i = _thumbViews.count; i < _columnCount; ++i) {
00133       TTThumbView* thumbView = [[[TTThumbView alloc] init] autorelease];
00134       [thumbView addTarget:self action:@selector(thumbTouched:)

00137       [_thumbViews addObject:thumbView];

00139         [self assignPhotoAtIndex:_photo.index+i toView:thumbView];
