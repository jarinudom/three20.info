00144   CGSize textSize = [_label.text sizeWithFont:_label.font];
00145 
00146   CGFloat indicatorSize = 0;
00147   [_activityIndicator sizeToFit];
00148   if (_activityIndicator.isAnimating) {
00149     if (_activityIndicator.height > textSize.height) {
00150       indicatorSize = textSize.height;
00151     } else {
00152       indicatorSize = _activityIndicator.height;
00153     }
00154   }
00155   
00156   CGFloat contentWidth = indicatorSize + kSpacing + textSize.width;
00157   CGFloat contentHeight = textSize.height > indicatorSize ? textSize.height : indicatorSize;
00158 
00159   if (_progressView) {
00160     [_progressView sizeToFit];
00161     contentHeight += _progressView.height + kSpacing;
00162   }
00163 
00164   CGFloat margin, padding, bezelWidth, bezelHeight;
00165   if (_style == TTActivityLabelStyleBlackBezel || _style == TTActivityLabelStyleWhiteBezel) {
00166     margin = kMargin;
00167     padding = kPadding;
00168     bezelWidth = contentWidth + padding*2;
00169     bezelHeight = contentHeight + padding*2;
00170   } else {
00171     margin = 0;
00172     padding = kBannerPadding;
00173     bezelWidth = self.width;
00174     bezelHeight = self.height;
00175   }
00176 
00177   CGFloat maxBevelWidth = TTScreenBounds().size.width - margin*2;
00178   if (bezelWidth > maxBevelWidth) {
00179     bezelWidth = maxBevelWidth;
00180     contentWidth = bezelWidth - (kSpacing + indicatorSize);
00181   }
00182   
00183   CGFloat textMaxWidth = (bezelWidth - (indicatorSize + kSpacing)) - padding*2;
00184   CGFloat textWidth = textSize.width;
00185   if (textWidth > textMaxWidth) {
00186     textWidth = textMaxWidth;
00187   }
00188       
00189   _bezelView.frame = CGRectMake(floor(self.width/2 - bezelWidth/2),
00190                                 floor(self.height/2 - bezelHeight/2),
00191                                 bezelWidth, bezelHeight);
00192   
00193   CGFloat y = padding + floor((bezelHeight - padding*2)/2 - contentHeight/2);
00194   
00195   if (_progressView) {
00196     if (_style == TTActivityLabelStyleBlackBanner) {
00197       y += kBannerPadding/2;
00198     }
00199     _progressView.frame = CGRectMake(kProgressMargin, y,
00200                                      bezelWidth - kProgressMargin*2, _progressView.height);
00201     y += _progressView.height + kSpacing-1;
00202   }

00208                                         indicatorSize, indicatorSize);
